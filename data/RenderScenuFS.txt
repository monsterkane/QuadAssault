uniform sampler2D geometrija;
uniform sampler2D lightmapa;
uniform vec3 ambijent;

void main()
{		
	vec4 boja;
	vec4 amb=vec4(ambijent,1.0);
	
	vec4 t1=texture2D(geometrija, gl_TexCoord[0].st);
	vec4 t2=texture2D(lightmapa, gl_TexCoord[0].st);
	
	boja=t1*amb;
        boja+=t1*t2*4.0;

	gl_FragColor = boja;
}
